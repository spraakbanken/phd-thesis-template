% The original todo command pushes everything to the side, which is not desirable
\newtoggle{lmargin}
\newcommand{\alternatingtodo}[2][]{
  \iftoggle{lmargin}
  {
    \todo[#1]{#2}
    \togglefalse{lmargin}
  }{
    {
      \let\marginpar\marginnote
      \reversemarginpar
      \todo[#1]{#2}
    }
    \toggletrue{lmargin}
  }
}
